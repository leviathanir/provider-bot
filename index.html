# Установка зависимостей: pip install fastapi uvicorn sqlalchemy alembic python-jose[cryptography] passlib[bcrypt] python-multipart pytest pytest-asyncio aiosqlite
"""
README
1. alembic init migrations; настроить sqlalchemy.url
2. alembic revision --autogenerate -m "init"; alembic upgrade head
3. uvicorn main:app --reload

import logging
from contextlib import asynccontextmanager
from datetime import datetime, timedelta
from typing import List, Optional
import re
import jwt
from fastapi import FastAPI, Depends, HTTPException, status, WebSocket, WebSocketDisconnect
from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
from pydantic import BaseModel, Field, validator
from sqlalchemy import create_engine, Column, Integer, String, DateTime, ForeignKey, Boolean, Text
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker, relationship, Session as SyncSession
from sqlalchemy.future import select
from sqlalchemy.sql import func
from passlib.context import CryptContext
import asyncio

# Настройка логирования
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# Конфигурация
SECRET_KEY = "supersecretkeyforsupportbot"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 60
DATABASE_URL = "sqlite+aiosqlite:///./support.db"

# ========== МОДЕЛИ SQLAlchemy (async) ==========
Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True, index=True)
    username = Column(String, unique=True, index=True, nullable=False)
    hashed_password = Column(String, nullable=False)
    sessions = relationship("Session", back_populates="user")

class Session(Base):
    __tablename__ = "sessions"
    id = Column(Integer, primary_key=True, index=True)
    session_id = Column(String, unique=True, index=True, nullable=False)  # UUID or string
    user_id = Column(Integer, ForeignKey("users.id"))
    created_at = Column(DateTime, server_default=func.now())
    user = relationship("User", back_populates="sessions")
    messages = relationship("Message", back_populates="session", cascade="all, delete")

class Message(Base):
    __tablename__ = "messages"
    id = Column(Integer, primary_key=True, index=True)
    session_id = Column(Integer, ForeignKey("sessions.id"))
    sender = Column(String)  # "user" или "bot"
    text = Column(Text)
    created_at = Column(DateTime, server_default=func.now())
    session = relationship("Session", back_populates="messages")

# ========== Pydantic схемы ==========
class UserCreate(BaseModel):
    username: str = Field(..., min_length=3, max_length=50)
    password: str = Field(..., min_length=4)

    @validator('username')
    def not_empty(cls, v):
        if not v.strip():
            raise ValueError('username пустой')
        return v

class UserOut(BaseModel):
    id: int
    username: str

class Token(BaseModel):
    access_token: str
    token_type: str

class SessionCreateResponse(BaseModel):
    session_id: str

class MessagePayload(BaseModel):
    text: str = Field(..., min_length=1)

class MessageOut(BaseModel):
    id: int
    sender: str
    text: str
    created_at: datetime

class HistoryOut(BaseModel):
    session_id: str
    messages: List[MessageOut]

# ========== Аутентификация ==========
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/auth/login")

def verify_password(plain_password, hashed_password):
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password):
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: timedelta = None):
    to_encode = data.copy()
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=15)
    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

async def get_current_user(token: str = Depends(oauth2_scheme), db: AsyncSession = Depends(get_db)):
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
    except jwt.PyJWTError:
        raise credentials_exception
    result = await db.execute(select(User).where(User.username == username))
    user = result.scalar_one_or_none()
    if user is None:
        raise credentials_exception
    return user

# ========== База данных ==========
async_engine = create_async_engine(DATABASE_URL, echo=True)
AsyncSessionLocal = sessionmaker(async_engine, class_=AsyncSession, expire_on_commit=False)

async def get_db():
    async with AsyncSessionLocal() as session:
        yield session

# ========== Логика бота (ключевые слова) ==========
def bot_reply(text: str) -> str:
    lower = text.lower()
    if re.search(r'привет|здравствуй|добрый', lower):
        return "Здравствуйте! Чем могу помочь?"
    if re.search(r'работа|функционал|можешь', lower):
        return "Я отвечаю на вопросы о продукте, помогаю с документацией и общими проблемами."
    if re.search(r'проблем|ошибк|баг|не работает', lower):
        return "Опишите проблему подробнее, я передам специалисту (или попробую помочь сам)."
    return "Спасибо за сообщение. Оператор скоро ответит."  # default

# ========== Эндпоинты FastAPI ==========
@asynccontextmanager
async def lifespan(app: FastAPI):
    # создание таблиц для простоты (без alembic в демо, но в README миграции)
    async with async_engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
    logger.info("Tables created (for demo, use Alembic in production)")
    yield

app = FastAPI(title="Support Bot API", lifespan=lifespan)

# Регистрация
@app.post("/auth/register", response_model=UserOut)
async def register(user: UserCreate, db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(User).where(User.username == user.username))
    if result.scalar_one_or_none():
        raise HTTPException(400, "Username exists")
    hashed = get_password_hash(user.password)
    new_user = User(username=user.username, hashed_password=hashed)
    db.add(new_user)
    await db.commit()
    await db.refresh(new_user)
    return new_user

# Логин
@app.post("/auth/login", response_model=Token)
async def login(form_data: OAuth2PasswordRequestForm = Depends(), db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(User).where(User.username == form_data.username))
    user = result.scalar_one_or_none()
    if not user or not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(400, "Incorrect username or password")
    access_token = create_access_token(data={"sub": user.username}, expires_delta=timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES))
    return {"access_token": access_token, "token_type": "bearer"}

# Создать сессию
@app.post("/chat/session", response_model=SessionCreateResponse)
async def create_session(current_user: User = Depends(get_current_user), db: AsyncSession = Depends(get_db)):
    import uuid
    session_id = str(uuid.uuid4())
    new_session = Session(session_id=session_id, user_id=current_user.id)
    db.add(new_session)
    await db.commit()
    return SessionCreateResponse(session_id=session_id)

# Получить историю (с проверкой принадлежности)
@app.get("/chat/history/{session_id}", response_model=HistoryOut)
async def get_history(session_id: str, current_user: User = Depends(get_current_user), db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(Session).where(Session.session_id == session_id))
    sess = result.scalar_one_or_none()
    if not sess:
        raise HTTPException(404, "Session not found")
    if sess.user_id != current_user.id:
        raise HTTPException(403, "Not your session")
    msgs_result = await db.execute(select(Message).where(Message.session_id == sess.id).order_by(Message.created_at))
    msgs = msgs_result.scalars().all()
    return HistoryOut(session_id=session_id, messages=[
        MessageOut(id=m.id, sender=m.sender, text=m.text, created_at=m.created_at) for m in msgs
    ])

# ========== WebSocket ==========
class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[WebSocket, dict] = {}

    async def connect(self, websocket: WebSocket, user: User, session: Session):
        await websocket.accept()
        self.active_connections[websocket] = {"user": user, "session": session}
        logger.info(f"WS connected user={user.username} session={session.session_id}")

    def disconnect(self, websocket: WebSocket):
        if websocket in self.active_connections:
            del self.active_connections[websocket]

    def get_session(self, websocket: WebSocket) -> Optional[Session]:
        info = self.active_connections.get(websocket)
        return info["session"] if info else None

manager = ConnectionManager()

@app.websocket("/ws/chat")
async def websocket_endpoint(websocket: WebSocket, token: str = None, db: AsyncSession = Depends(get_db)):
    if not token:
        await websocket.close(code=1008, reason="Token missing")
        return
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username = payload.get("sub")
        if not username:
            raise ValueError
    except:
        await websocket.close(code=1008, reason="Invalid token")
        return

    result = await db.execute(select(User).where(User.username == username))
    user = result.scalar_one_or_none()
    if not user:
        await websocket.close(code=1008, reason="User not found")
        return

    # ожидаем первое сообщение с session_id
    try:
        data = await websocket.receive_json()
        if data.get("type") != "bind_session" or not data.get("session_id"):
            await websocket.close(code=1008, reason="First message must be bind_session with session_id")
            return
        session_id = data["session_id"]
        sess_res = await db.execute(select(Session).where(Session.session_id == session_id))
        session = sess_res.scalar_one_or_none()
        if not session or session.user_id != user.id:
            await websocket.close(code=1008, reason="Invalid session or not yours")
            return
    except Exception as e:
        await websocket.close(code=1011, reason="Handshake error")
        return

    await manager.connect(websocket, user, session)

    try:
        while True:
            msg_data = await websocket.receive_json()
            if msg_data.get("type") == "message":
                text = msg_data.get("text", "").strip()
                if not text:
                    continue
                # сохраняем сообщение пользователя
                sess_obj = manager.get_session(websocket)
                if not sess_obj:
                    continue
                user_msg = Message(session_id=sess_obj.id, sender="user", text=text)
                db.add(user_msg)
                # ответ бота
                reply = bot_reply(text)
                bot_msg = Message(session_id=sess_obj.id, sender="bot", text=reply)
                db.add(bot_msg)
                await db.commit()
                # отправляем оба сообщения клиенту (можно только бота, но покажем оба)
                await websocket.send_json({"role": "user", "text": text})
                await websocket.send_json({"role": "bot", "text": reply})
    except WebSocketDisconnect:
        manager.disconnect(websocket)
        logger.info("WebSocket disconnected")
